<DocPage fieldName="down" fieldType="function" spec={Spec} />

```scss
@use '@sass-fairy/break' with (
  $vertical-sizes: $your-sizes
);

// 1. As a function
@debug break.down('md');
  // '(max-height: 767.98px)'

@debug break.down(800px);
  // '(max-height: 799.98px)'

// 2. As a mixin
body {
  padding: 2rem;

  @include break.down('md') {
    padding: 1rem;
  }

  @include break.down(800px) {
    padding: 1rem;
  }
}
  // body { padding: 2rem }
  // @media (max-height: 767.98px) {
  //   body { padding: 1rem }
  // }
  // @media (max-height: 799.98px) {
  //   body { padding: 1rem }
  // }
```
