```scss
@use '@sass-fairy/list';
@use '@sass-fairy/map';

// 1. As a call
@debug map.compare-numeric-value(b 8em, f 10mm);
  // -2

// 2. As a reference
$map: (b: 8em, f: 10mm, d: a, a: 3, c: 1cm, e: 2mm);

@debug list.sort($map, map.compare-numeric-value());
  // c 1cm, e 2mm, a 3, b 8em, f 10mm, d a
```
